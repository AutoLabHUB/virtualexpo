<script>
  // Smooth scroll + TOC active states
  const tocLinks = document.querySelectorAll('.toc-link');
  const sections = Array.from(document.querySelectorAll('section.card.section'));

  tocLinks.forEach(link => {
    link.addEventListener('click', (e) => {
      e.preventDefault();
      const targetId = link.getAttribute('data-target');
      const section = document.getElementById(targetId);
      if (section) {
        window.scrollTo({
          top: section.getBoundingClientRect().top + window.scrollY - 20,
          behavior: 'smooth'
        });
      }
    });
  });

  function updateActiveToc() {
    let currentId = null;
    const offset = 90;
    for (const section of sections) {
      const rect = section.getBoundingClientRect();
      if (rect.top - offset <= 0 && rect.bottom > offset) {
        currentId = section.id;
        break;
      }
    }
    tocLinks.forEach(link => {
      const targetId = link.getAttribute('data-target');
      link.classList.toggle('active', targetId === currentId);
    });
  }

  document.addEventListener('scroll', () => {
    updateActiveToc();
  }, { passive: true });

  updateActiveToc();

  // Modal handling
  const modal = document.getElementById('accept-modal');

  function openAcceptModal() {
    modal.classList.add('show');
  }

  function closeAcceptModal() {
    modal.classList.remove('show');
  }

  modal.addEventListener('click', (e) => {
    if (e.target === modal) {
      closeAcceptModal();
    }
  });

  // Build mailto link from selected option + comments
  function buildMailtoLink() {
    const selectedOption = document.querySelector('input[name="proposal_option"]:checked');
    const comments = document.getElementById('proposal_comments').value.trim();
    const nameOrg = document.getElementById('proposal_name').value.trim();

    const subject = encodeURIComponent('Acceptance: Virtual Expo Platform Proposal');
    let bodyLines = [];

    bodyLines.push('Hi Nex-a team,');
    bodyLines.push('');
    bodyLines.push('We would like to accept the Virtual Expo Platform proposal with the following details:');
    bodyLines.push('');
    if (selectedOption) {
      bodyLines.push('Selected option: ' + selectedOption.value);
    }
    if (nameOrg) {
      bodyLines.push('Contact name / organisation: ' + nameOrg);
    }
    if (comments) {
      bodyLines.push('');
      bodyLines.push('Notes / questions:');
      bodyLines.push(comments);
    }
    bodyLines.push('');
    bodyLines.push('Please let us know the next steps to proceed.');
    bodyLines.push('');
    bodyLines.push('Kind regards,');
    if (nameOrg) {
      bodyLines.push(nameOrg);
    }

    const body = encodeURIComponent(bodyLines.join('\n'));
    const href = 'mailto:hello@nex-a.com.au?subject=' + subject + '&body=' + body;

    const link = document.getElementById('mailto-link');
    link.setAttribute('href', href);
    // default anchor behaviour will open the mail client
  }

  // Cost calculator logic (table-based)
  const featureRows = document.querySelectorAll('.calc-row-feature');
  const summaryMainEl = document.getElementById('calc-summary-main');
  const summaryExtraEl = document.getElementById('calc-summary-extra');
  const breakdownEl = document.getElementById('calc-breakdown');

  function formatMoney(value) {
    const rounded = Math.round(value);
    return 'A$' + rounded.toString().replace(/\B(?=(\d{3})+(?!\d))/g, ',');
  }

  function recalcCalculator() {
    let minTotal = 0;
    let maxTotal = 0;
    const breakdownLines = [];

    featureRows.forEach(row => {
      const min = Number(row.dataset.min || '0');
      const max = Number(row.dataset.max || '0');
      const checkbox = row.querySelector('.calc-feature-checkbox');
      const labelCell = row.cells[1];
      if (!checkbox || !labelCell) return;

      const label = labelCell.textContent.trim();

      if (checkbox.checked || checkbox.disabled) {
        minTotal += min;
        maxTotal += max;
        breakdownLines.push(
                '<div class="calc-breakdown-line"><span>' + label +
                '</span><span>' + formatMoney(min) + ' â€“ ' + formatMoney(max) +
                '</span></div>'
        );
      }
    });

    summaryMainEl.textContent = formatMoney(minTotal) + ' â€“ ' + formatMoney(maxTotal);
    summaryExtraEl.textContent = 'Indicative total for the selected feature set (ex GST).';
    breakdownEl.innerHTML = breakdownLines.join('');
  }

  featureRows.forEach(row => {
    const checkbox = row.querySelector('.calc-feature-checkbox');
    if (checkbox && !checkbox.disabled) {
      checkbox.addEventListener('change', recalcCalculator);
    }
  });

  recalcCalculator();
</script>
